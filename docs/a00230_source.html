63 volatile uint16_t appPowerSupply_Execute(void)
64 {
65  volatile uint16_t retval=1;
66 
67  // Capture data values
68  buck.data.v_in = BUCK_VIN_ADCBUF;
69  buck.data.i_sns[0] = BUCK_ISNS_ADCBUF;
70  buck.data.temp = BUCK_TEMP_ADCBUF;
71 
72  // Average inductor current value
73  isns_samples += buck.data.i_sns[0];
74  if(!(++_isns_sample_count & ISNS_AVG_BITMASK))
75  {
76  buck.data.i_out = (isns_samples >> 3);
77  isns_samples = 0;
78  }
79 
80  // Execute buck converter state machine
81  retval &= drv_BuckConverter_Execute(&buck);
82 
83  // Buck regulation error is only active while controller is running
84  // and while being tied to a valid reference
85  if( (buck.state_id.bits.opstate_id >= BUCK_OPSTATE_RAMPUP) &&
86  (buck.state_id.bits.substate_id >= BUCK_OPSTATE_V_RAMP_UP) )
87  {
88  fltobj_BuckRegErr.ReferenceObject.ptrObject = buck.v_loop.controller->Ports.ptrControlReference;
89  #if (PLANT_MEASUREMENT == false)
90  fltobj_BuckRegErr.Status.bits.Enabled = buck.v_loop.controller->status.bits.enabled;
91  #endif
92  }
93  else {
94  fltobj_BuckRegErr.Status.bits.Enabled = false;
95  }
96 
97  return(retval);
98 }
